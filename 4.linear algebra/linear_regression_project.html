def matxMultiply(A, B):
    if len(A[0])!=len(B):
        raise ValueError('Shape error')
    a = [None] * len(A)
    Btras = transpose(B)
    lBtras = len(Btras)
    for key,val in enumerate(A):
       b = [None] * lBtras
       for k,v in enumerate(Btras):
           b[k] = sum([x*y for x,y in zip(val,v)])
       a[key] = b
    return a

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""

def gj_Solve(A, b, decPts=4, epsilon = 1.0e-16):
    if len(A) != len(b):
        return None
    #Ab
    Ab=augmentMatrix(A,b)

    row, col=shape(A)
    for c in range(col):
        #maxcol
        maxValueRowNumber = c
        maxAbs = abs(Ab[c][c])
        
        #
        for i in range(c+1, row):
            if abs(Ab[i][c]) > maxAbs:
                maxAbs = abs(Ab[i][c])
                maxValueRowNumber = i
        if maxAbs < epsilon:
            #0
            return None
        else:
            #c
            if maxValueRowNumber!=c:
                swapRows(Ab, c, maxValueRowNumber)

            #c1
            scaleRow(Ab, c, 1.0 / Ab[c][c])

            #c0
            for r in range(row):
                if r!=c:
                    addScaledRow(Ab, r, c, -Ab[r][c])

    matxRound(Ab, decPts)
    resultVector = []
    for c in range(row):
        resultVector.append([Ab[c][-1]])

    return resultVector

def calculateMSE2D(X,Y,m,b):
    MSE = 0
    sum_squared = 0
    n = len(X)
    for x,y in zip(X,Y):
        sum_squared = sum_squared + (y -  m*x - b)**2
